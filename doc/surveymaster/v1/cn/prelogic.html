		    ]
	    },

    "conditions" : [/* 每一个数组元素就是一个条件，组内各条件都是 AND 的关系 */

		    "question_id" : 7,
		    "range" : [1,3]
	    },
	    {
		    "question_id" : 8,
		    "range" : [2,-1],
		    "type" : "all/any/none"
	    },
	    {
		    "question_id" : 9,
		    "range" : [4,5],
		    "type" : "all/any/none"
	    },
	    {
		    "question_id" : 10,

	    },
	    {
		    "question_id" : 11,
		    "min" : 6
		    "max" : 10
	    },
	    {
		    "question_id" : 12,
		    "include" : ["99click","好"],
		    "exclude" : ["不好"]

    ]
