获取指定媒体频道列表

GET /medias/:media_id/channels

响应

Status: 200 OK
Link: <http://track.admasterapi.com/channels?page=2>; rel="next",
      <http://track.admasterapi.com/channels?page=10>; rel="last"
X-RateLimit-Limit: 5000
X-RateLimit-Remaining: 4999

[
   {
        //频道ID，全局唯一
        "id": 1025,
        //频道名称
        "name": "体育新闻",
        //`webpage` 网页, `video` 视频广告, `client` 客户端, `se` 搜索引擎, `email` 邮件, `other` 其他
        "type": "web",
        //广告位在第几屏幕
        "screen": 3,
        //频道地址
        "home": "http://www.admaster.com.cn/",
        //物料类型 `flash`，`image`，`video`, `textlink`, `other` 默认：`flash`
        "material_type": 'flash',

        "material_dimension": "400x300",

        "material_size": 200,

]
